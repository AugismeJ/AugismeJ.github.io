		class Player(object):
    def __init__(self,hp = 100, food = 100, water = 100, o2 = 100):
        self.HP = hp
        self.inventory = Inventory()
        self.food = food
        self.water = water
        self.o2 = o2
    def __str__(self):
        r = "HP: " + str(self.HP) + "\n" + "Food: " + str(self.food) + 
				"\n" + "Water: " + str(self.water) + "\n" + "O2: "+str(self.o2)
        return r
    def damage(self, damage = 1, deathMessage = "You Died"):
        self.HP -= damage
        if self.HP <= 0:
            print(deathMessage)
            self.die()
    def die(self):
        self.inventory.clear()
        self.food = 100
        self.water = 100
        self.o2 = 100
        self.HP = 100
